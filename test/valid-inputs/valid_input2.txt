// Test arithmetic operations
int a = 15;
int b = 3;
int result;

result = a + b;    // Addition
print result;      // Should print 18

result = a - b;    // Subtraction
print result;      // Should print 12

result = a * b;    // Multiplication
print result;      // Should print 45

result = a / b;    // Division
print result;      // Should print 5

// Test boolean operations
bool test = true;
bool other = false;
bool final;

final = test && other;  // AND operation
print final;            // Should print false

final = test || other;  // OR operation
print final;            // Should print true
